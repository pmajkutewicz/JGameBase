415 /**
416  * Tests if this string starts with the specified prefix.
417  * @returns <code>true</code> if the character sequence represented by the
418  *          argument is a prefix of the character sequence represented by
419  *          this string; <code>false</code> otherwise.
420  */
421 String.prototype.startsWith = function(/**String*/ needle) /**boolean*/ {
422 	if (arguments.length !== 1) {
423 		throw "ArgumentCountError: Call to method 'String.startsWith()' with wrong number of arguments!";
424 	}
425 	if (typeof arguments[0] !== "string") {
426 		throw "TypeError: Call to method 'String.startsWith()' with wrong type of argument (parameter 'needle' must be of type string)!";
427 	}
428 
429 	if (needle.length > this.length) {
430 		return false;
431 	}
432 	return (this.indexOf(needle) === 0);
433 };
434 
435 /**
436  * Tests if this string ends with the specified suffix.
437  * @returns <code>true</code> if the character sequence represented by the
438  *          argument is a suffix of the character sequence represented by
439  *          this object; <code>false</code> otherwise.
440  */
441 String.prototype.endsWith = function(/**String*/ needle) /**boolean*/ {
442 	if (arguments.length !== 1) {
443 		throw "ArgumentCountError: Call to method 'String.endsWith()' with wrong number of arguments!";
444 	}
445 	if (typeof arguments[0] !== "string") {
446 		throw "TypeError: Call to method 'String.endsWith()' with wrong type of argument (parameter 'needle' must be of type string)!";
447 	}
448 
449 	if (needle.length > this.length) {
450 		return false;
451 	}
452 	return (this.lastIndexOf(needle) === (this.length - needle.length));
453 };

455 /**
456  * Returns a copy of the string, with leading and trailing space omitted.
457  * @return  A copy of this string with leading and trailing space removed,
458  *          or this string if it has no leading or trailing space.
459  */
460 String.prototype.trim = function() /**String*/ {
461 	if (arguments.length !== 0) {
462 		throw "ArgumentCountError: Call to method 'String.trim()' with wrong number of arguments!";
463 	}
464 
465 	var temp = this.toString();
466 
467 	while (temp.startsWith(" ")) {
468 		temp = temp.substring(1, temp.length);
469 	}
470 
471 	while (temp.endsWith(" ")) {
472 		temp = temp.substring(0, temp.length - 1);
473 	}
474 
475 	return temp;
476 };

478 /**
479  * Returns <code>true</code> if and only if this string contains one or more of the specified
480  * values. <p>
481  * For boolean values of <code>true</code> the String is compared with "yes", "on",
482  * "true", and "1"; For boolean values of <code>false</code> with "no", "off",
483  * "false", and "0".<p>
484  * For String values the following wildcards are supported:<ul>
485  * <li>"*"</li> The String is not empty
486  * <li>"value"</li> The String must match the value
487  * <li>"value*"</li> The String must start with the value
488  * <li>"*value"</li> The String must end with the value
489  * <li>"*value*"</li> The String must contain the value
490  * </ul><p>
491  * E.g. to check if the value of the variable "command" is set<p><pre>
492  * if (command.contains("*")) {
493  *   ...
494  * }</pre>
495  *
496  * @param {boolean or String} ... the values to search for
497  * @return <code>true</code> if this string contains one or more of the specified
498  *         values, <code>false</code> otherwise
499  */
500 String.prototype.contains = function() /**boolean*/ {
501 	if (arguments.length === 0) {
502 		throw "ArgumentCountError: Call to method 'String.contains()' with wrong number of arguments!";
503 	}
504 
505 	var found = false, haystack = this.toLowerCase();
506 
507 	for ( var i = 0; i < arguments.length; i++) {
508 
509 		if (typeof arguments[i] === "boolean") {
510 			if (arguments[i]) {
511 				return this.contains("yes", "on", "true", "1");
512 			} else {
513 				return this.contains("no", "off", "false", "0");
514 			}
515 		}
516 
517 		if (typeof arguments[i] !== "string") {
518 			throw "TypeError: Call to method 'String.contains()' with wrong type of argument (parameters must be of type string or boolean)!";
519 		}
520 
521 		var needle = arguments[i].toLowerCase();
522 
523 		if (needle === "*") {
524 			return (haystack !== "");
525 		}
526 
527 		var firstIsStar = needle.startsWith("*");
528 		var lastIsStar = needle.endsWith("*");
529 
530 		if ((firstIsStar) && (lastIsStar) && (!found)) {
531 			// *needle*
532 			needle = needle.substring(1, needle.length - 1);
533 			found = (haystack.indexOf(needle) !== -1);
534 		} else if ((!firstIsStar) && (lastIsStar) && (!found)) {
535 			// needle*
536 			needle = needle.substring(0, needle.length - 1);
537 			found = haystack.startsWith(needle);
538 		} else if ((firstIsStar) && (!lastIsStar) && (!found)) {
539 			// *needle
540 			needle = needle.substring(1, needle.length);
541 			found = haystack.endsWith(needle);
542 		} else if ((!firstIsStar) && (!lastIsStar) && (!found)) {
543 			// needle
544 			found = (haystack === needle);
545 		}
546 	}
547 
548 	return found;
549 };

566 /**

651 /**
652  * @ignore
653  * Internal implementation, should NOT be called directly.
654  */
655 function _Set_CFG_Item(filename, key, delimiter, value) {
656 	var reader;
657 	var writer;
658 
659 	var found = false;
660 	var newLine = key + delimiter + value;
661 
662 	try {
663 		var inFile = new File(filename);
664 		var outFile = new File(filename + "_new");
665 
666 		reader = new java.io.BufferedReader(new java.io.FileReader(inFile));
667 		writer = new java.io.BufferedWriter(new java.io.FileWriter(outFile));
668 
669 		// ... Loop as long as there are input lines.
670 		var line = null;
671 		while ((line = reader.readLine()) !== null) {
672 			var lowerLine = line.toLowerCase();
673 			var lowerKey = key.toLowerCase();
674 
675 			if ((lowerLine.startsWith(lowerKey + delimiter))
676 					|| (lowerLine.startsWith(lowerKey + " " + delimiter))
677 					|| (lowerLine.startsWith(lowerKey + "\t" + delimiter))) {
678 				writer.write(newLine); // write new line
679 				found = true;
680 			} else {
681 				writer.write(line); // copy line
682 			}
683 			writer.newLine(); // Write system dependent end of line.
684 		}
685 
686 		if (!found) { // add line at end of file
687 			writer.write(newLine); // write new line
688 			writer.newLine(); // Write system dependent end of line.
689 		}
690 
691 		inFile.remove();
692 		outFile.renameTo(inFile);
693 
694 	} catch (error) {
695 		println(error.name + ": " + error.message);
696 	} finally {
697 		if (reader !== null) {
698 			try {
699 				reader.close();
700 			} catch (ignore1) {
701 			}
702 		}
703 		if (writer !== null) {
704 			try {
705 				writer.close();
706 			} catch (ignore2) {
707 			}
708 		}
709 	}
710 }
